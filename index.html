    }

void aff_ligne_carre_1(int n, int c) { // définit la fonction en donnant n et c comme valeurs entières
    int col = 0; 
    while (col < n) { // (on vérifie qu'on a pas dépassé la dernière colonne)
        printf("%d", c); // printf("%d", ...) permet d'afficher une valeur entière (ici c)
        col = col + 1; // (on ajoute 1 à la variable col)
    }
    putchar('\n'); // (on retourne à la ligne)
}

#include <stdio.h>

void carre_3(int);

void carre_3(int n) {
    int row = 0, col;
    while (row < n) {
        int bh;
        bh = (row == n-1) || (row == 0); // || veut dire 'ou' et && veut dire 'et'
        col= 0;
        while(col < n) {
            int bv;
            bv = (col == 0) || (col == n-1);
            if(bh || bv) {
                putchar('*');
            }
            else {
                putchar(' ');
            }
            col = col + 1;
        }
        putchar('\n');
        row = row + 1;
    }
}

carre_4 :


#include <stdio.h>

void carre_4(int n) {
    int row = 0, col;
    while (row < n) {
        int bh;
        bh = (0 == row) || ((n-1) == row);
        col = 0;
        while(col < n) {
            int bv, d1, d2;
            bv = (0 == row) || ((n-1) == row);
            d1 = col == row;
            d2 = col == (n-1-row);
            if(bh) {
                if(bv) {
                    putchar('+');
                }
                else {
                    putchar('-');
                }
            }
            else {
                if(bv) {
                    putchar('|');
                }
                else {
                    if(d1) {
                        if(d2) {
                            putchar('x');
                        }
                        else {
                            putchar('\\') // "\" est utilisé pour des caractère spéciaux on doit donc mettre "\" devant pour lui dire qu'on veut l'écrire et pas faire un caractère spécial (comme ('\n'))
                        }
                    }
                    else {
                        if (d2) {
                            putchar('/');
                        }
                        else {
                            putchar(' ');
                        }
                    }
                }   
            }                      
        }
        col = col + 1;
        putchar('\n');
        row = row + 1;
    }
}

    return 0;
    }

void roi (int n) {
    int row = 0, col;
    while(row < n) {
        col = 0;
        while(col < n) {
            printf("%d", lemax(col, row));
            col = col + 1;
        }
        putchar('\n');
        row = row + 1;
    }
}
int lemax(int x, int y) {
    if(x < y) {
        return y;
    }
    return x;
}
